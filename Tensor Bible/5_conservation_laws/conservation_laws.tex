\section{Constitutive Variables}

\begin{itemize}
	\item $\rho_{0} = \rho_{0}(\vb{X})$: density in material frame
	\item $\rho = \rho(\vb{x}, t)$: density in spatial frame
	\item $\vb{S} = \vb{S}(\vb{X}, t)$: second Piola-Kirchoff stress tensor in material frame
	\begin{itemize}
		\item $\vb{S} := J \vb{F}^{-1} \vb*{\sigma} \vb{F}^{-T}$
	\end{itemize}
	\item $\vb{P} = \vb{P}(\vb{X}, t)$: first Piola-Kirchoff stress tensor in material frame
	\begin{itemize}
	\item $\vb{P} := J \vb*{\sigma} \vb{F}^{-T} = \vb{F} \vb{S}$
	\end{itemize}
	\item $\vb*{\sigma} = \vb*{\sigma}(\vb{x}, t)$: Cauchy stress tensor in spatial frame
	\item $\vb{B} = \vb{B}(\vb{X}, t)$: body force per unit mass in material frame
	\item $\vb{b} = \vb{b}(\vb{x}, t)$: body force per unit mass in spatial frame
	\item $E = E(\vb{X}, t)$: internal energy per unit mass in material frame
	\item $e = e(\vb{x}, t)$: internal energy per unit mass in spatial frame
	\item $I = I(\vb{X}, t)$: potential? energy per unit mass in material frame
	\item $i = i(\vb{x}, t)$: potential? energy per unit mass in spatial frame
	\item $\vb{Q} = \vb{Q}(\vb{X}, t)$: heat flux in material frame
	\item $\vb{q} = \vb{q}(\vb{x}, t)$: heat flux in spatial frame
	\item $\Theta = \Theta(\vb{X}, t)$: temperature in material frame
	\item $\theta = \theta(\vb{x}, t)$: temperature energy per unit mass in spatial frame
\end{itemize}

\section{Reynold's Transport Theorem}

\begin{flalign}
	&& \dv{t} \int_{\vb{x} \in V(t)} \vb{q} \dd{\Omega_{t}} &= \int_{\vb{x} \in V(t)} \pdv{\vb{q}}{t} \dd{\Omega_{t}} + \int_{\vb{x} \in \partial \left(V(t)\right)} \vb{q} \left( \vb{v} \vdot \vb{\hat{n}} \right) \dd{\partial \Omega_{t}} & \text{Proof \ref{proof:}} \label{equation:}
\end{flalign}

\section{Conservation of Mass}

\subsection{Material Frame}

\begin{flalign}
	&& \rho_{0} &= \rho \circ \phi_{t} J & \text{Proof \ref{proof:com_mf_dif}} \label{equation:com_mf_dif}
\end{flalign}

\subsection{Spatial Frame}

\begin{flalign}
	&& \dv{t} \int_{\vb{x} \in V \subseteq \Omega_{t}} \rho(\vb{x}, t) \dd{V} + \int_{\vb{x} \in \partial V} \rho(\vb{x}, t) \vb{v}(\vb{x}, t) \vdot \hat{\vb{n}}(\vb{x}) \dd{(\partial  V)} &= 0 & \text{Integral form. Proof \ref{proof:com_sf_int}} \label{equation:com_sf_int} \\
	&& \pdv{\rho}{t} + \div{\rho \vb{v}} &= 0 & \text{Proof \ref{proof:com_sf_dif}} \label{equation:com_sf_dif}
\end{flalign}

\section{Conservation of Momentum}

\subsection{Material Frame}

\begin{flalign}
	&& \rho_{0} \dv{V}{t} &= \Div_{\vb{X}} \vb{P} + \rho_{0} \vb{B} & \text{Proof \ref{proof:cop_mf_dif}} \label{equation:cop_mf_dif} \\
\end{flalign}

\subsection{Spatial Frame}

\begin{flalign}
	&& \pdv{\left( \rho \vb{v} \right)}{t} + \Div{\left( \rho \vb{v} \otimes \vb{v} \right)} &= \Div{\vb*{\sigma}} + \rho \vb{b} & \text{Conservative form. Proof \ref{proof:cop_sf_dif_cons}} \label{equation:cop_sf_dif_cons} \\
	&& \rho \left(\pdv{\vb{v}}{t} + \left( \Grad{\vb{v}} \right) \vb{v}\right) &= \Div{\vb*{\sigma}} + \rho \vb{b} & \text{Reduced form. Proof \ref{proof:cop_sf_dif_red}} \label{equation:cop_sf_dif_red} \\
\end{flalign}

\section{Conservation of Angular Momentum}

\subsection{Material Frame}

\begin{flalign}
	&& \vb{S} &= \vb{S}^{T} & \text{Reduced form. Proof \ref{proof:col_mf_dif_red}} \label{equation:col_mf_dif_red} \\
\end{flalign}

\subsection{Spatial Frame}

\begin{flalign}
	&& \pdv{\left( \vb{x} \cross \rho \vb{v} \right)}{t} + \Div{\left( \vb{x} \cross \rho \vb{v} \otimes \vb{v} \right)} &= \Div{\left(\vb{x} \cross \vb*{\sigma}\right)} + \vb{x} \cross \rho \vb{b} & \text{Conservative form. Proof \ref{proof:col_sf_dif_cons}} \label{equation:col_sf_dif_cons} \\
	&& \vb*{\sigma} &= \vb*{\sigma}^{T} & \text{Reduced form. Proof \ref{proof:col_sf_dif_red}} \label{equation:col_sf_dif_red} \\
\end{flalign}

\section{Conservation of Energy}

\subsection{Material Frame}

\begin{flalign}
	&& \rho_{0} \dv{I}{t} + \Div_{\vb{X}} \vb{Q} &= \rho_{0} \vb{R} + \vb{S} \vddot \vb{D} & \text{Proof \ref{proof:tot_coe_mf_dif}} \label{equation:tot_coe_mf_dif} \\
\end{flalign}

\subsection{Spatial Frame}

\begin{flalign}
	&& \pdv{\left( \rho e \right)}{t} + \Div{\left( \rho e \vb{v} \right)} &= \Div{\left( \vb*{\sigma} \vb{v} \right)}  + \rho r + \rho \vb{b} \vdot \vb{b} - \Div{\vb{q}} & \text{Conservation form. Proof \ref{proof:coe_sf_dif_cons}} \label{equation:coe_sf_dif_cons} \\
	&& \rho \left( \pdv{e}{t} + \Grad{e} \vdot \vb{v} \right) &= \vb*{\sigma} \vddot \vb{d}  + \rho r + \rho \vb{b} \vdot \vb{b} - \Div{\vb{q}} & \text{Reduced form. Proof \ref{proof:coe_sf_dif_red}} \label{equation:coe_sf_dif_red} \\
\end{flalign}

\section{Entropy Inequality}

\subsection{Material Frame}

\begin{flalign}
	&& \rho_{0} \dv{N}{t} + \Div_{\vb{X}}\frac{\vb{Q}}{\Theta} &\geq \rho_{0} \frac{R}{\Theta} & \text{Proof \ref{proof:tot_ion_mf_dif}} \label{equation:tot_ion_mf_dif}
\end{flalign}

\subsection{Spatial Frame}

\begin{flalign}
	&& \pdv{\left( \rho \eta \right)}{t} + \Div{\left( \rho \eta \vb{v} \right)} + \Div{\frac{\vb{q}}{\Theta}} - \rho r &\geq 0 & \text{Proof \ref{proof:tot_ion_sf_dif}} \label{equation:tot_ion_sf_dif} \\
\end{flalign}
