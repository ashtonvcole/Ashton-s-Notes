\begin{itemize}
	\item $\varrho, \rho$: density in material/spatial frames
	\item $\vb*{\Sigma}, \vb*{\sigma}$: Cauchy stress tensor in material/spatial frames
	\item $\vb{S}$: second Piola-Kirchoff stress tensor in material frame
	\begin{itemize}
		\item $\vb{S} := J \vb{F}^{-1} \vb*{\Sigma} \vb{F}^{-T}$
	\end{itemize}
	\item $\vb{P} = \vb{P}(\vb{X}, t)$: first Piola-Kirchoff stress tensor in material frame
	\begin{itemize}
		\item $\vb{P} := J \vb*{\Sigma} \vb{F}^{-T} = \vb{F} \vb{S}$
	\end{itemize}
	\item $\vb{B}, \vb{b}$: body force per unit mass in material/spatial frames
	\item $E, e$: energy per unit mass in material frame
	\item $I, i$: internal energy per unit mass in material frame
	\item $i = i(\vb{X}, t)$: potential? energy per unit mass in spatial frame
	\item $\vb{Q}, \vb{q}$: heat flux in material/spatial frames
\end{itemize}

\section{Conservation of Momentum}

\begin{itemize}
	\item Quantity: momentum, $\vb{Q} = \varrho \vb{V}, \vb{q} = \rho \vb{v}$
	\item Surface fluxes: Cauchy stress, $\vb*{\Phi} = -\vb*{\Sigma}, \vb*{\phi} = -\vb*{\sigma}$
	\item Internal sources: body forces, $\vb{R} = \varrho \vb{B}, \vb{r} = \rho \vb{b}$
\end{itemize}

\subsection{Material Frame}

\begin{flalign}
	&& \rho_{0} \dv{V}{t} &= \Div_{\vb{X}} \vb{P} + \rho_{0} \vb{B} & \text{Proof \ref{proof:cop_mf_dif}} \label{equation:cop_mf_dif} \\
\end{flalign}

\subsection{Spatial Frame}

\begin{flalign}
	&& \pdv{\left( \rho \vb{v} \right)}{t} + \Div{\left( \rho \vb{v} \otimes \vb{v} \right)} &= \Div{\vb*{\sigma}} + \rho \vb{b} & \text{Conservative form. Proof \ref{proof:cop_sf_dif_cons}} \label{equation:cop_sf_dif_cons} \\
	&& \rho \left(\pdv{\vb{v}}{t} + \left( \Grad{\vb{v}} \right) \vb{v}\right) &= \Div{\vb*{\sigma}} + \rho \vb{b} & \text{Reduced form. Proof \ref{proof:cop_sf_dif_red}} \label{equation:cop_sf_dif_red} \\
\end{flalign}

\section{Conservation of Energy}

\begin{itemize}
	\item Quantity: energy, $Q = E, q = e$
	\item Surface fluxes: Cauchy power and heat flux, $\vb*{\Phi} = -\vb{V} \vdot \vb*{\Sigma} + \vb{Q}, \vb*{\phi} = -\vb{v} \vdot \vb*{\sigma} + \vb{q}$
	\item Internal sources: body force power and generic energy sources, $R =  \varrho \vb{B} \vdot \vb{B}  + \varrho R, r =  \rho \vb{b} \vdot \vb{v}  + \rho r$
\end{itemize}

\subsection{Material Frame}

\begin{flalign}
	&& \rho_{0} \dv{I}{t} &= \vb{S} \vddot \vb{D} - \Div_{\vb{X}} \vb{Q} + \rho_{0} \vb{R} & \text{Proof \ref{proof:coe_mf_dif_red}} \label{equation:tot_coe_mf_dif_red}
\end{flalign}

\subsection{Spatial Frame}

\begin{flalign}
	&& \pdv{\left( \rho e \right)}{t} + \Div{\left( \rho e \vb{v} \right)} &= \Div{\left( \vb{v} \vdot \vb*{\sigma} - \vb{q} \right)} + \rho \vb{b} \vdot \vb{v}  + \rho r & \text{Conservative form. Proof \ref{proof:coe_sf_dif_cons}} \label{equation:coe_sf_dif_cons} \\
	&& \rho \left( \pdv{e}{t} + \Grad{e} \vdot \vb{v} \right) &= \vb*{\sigma} \vddot \vb{d} - \Div{\vb{q}} + \rho \vb{b} \vdot \vb{v}  + \rho r & \text{Reduced form. Proof \ref{proof:coe_sf_dif_red}} \label{equation:coe_sf_dif_red} \\
\end{flalign}

\section{Entropy Inequality}

\begin{itemize}
	\item Quantity: entropy, $Q = N, q = \eta$
	\item Surface fluxes: entropy from heat flux, $\vb*{\Phi} = \frac{\vb{Q}}{\Theta}, \vb*{\phi} = \frac{\vb{q}}{\theta}$
	\item Internal sources: entropy generation from energy sources, $R =  \varrho \frac{R}{\Theta}, r =  \rho \frac{r}{\theta}$
\end{itemize}

\subsection{Material Frame}

\begin{flalign}
	&& \rho_{0} \dv{N}{t} + \Div_{\vb{X}}\frac{\vb{Q}}{\Theta} - \rho_{0} \frac{R}{\Theta} &\geq 0 & \text{Proof \ref{proof:tot_ion_mf_dif}} \label{equation:ion_mf_dif}
\end{flalign}

\subsection{Spatial Frame}

\begin{flalign}
	&& \pdv{\left( \rho \eta \right)}{t} + \Div{\left( \rho \eta \vb{v} \right)} + \Div{\frac{\vb{q}}{\Theta}} - \rho \frac{r}{\theta} &\geq 0 & \text{Conservative form. Proof \ref{proof:ion_sf_dif_cons}} \label{equation:ion_sf_dif_cons} \\
	&& \rho \left( \pdv{\eta}{t} + \Grad{\eta} \vdot \vb{v} \right) + \Div{\frac{\vb{q}}{\Theta}} - \rho \frac{r}{\theta} &\geq 0 & \text{Reduced form. Proof \ref{proof:ion_sf_dif_red}} \label{equation:ion_sf_dif_red}
\end{flalign}

\end{itemize}

	&& \pdv{c}{t} + \div{\left( \vb{v} c - \vb{D} \grad{c} \right)} &= R & \text{Proof \ref{proof:}} \label{equation:adv_dif_rea_scal} \\
