\begin{itemize}
	\item $\varrho, \rho$: density in material/spatial frames
	\item $\vb*{\Sigma}, \vb*{\sigma}$: Cauchy stress tensor in material/spatial frames
	\item $\vb{S}$: second Piola-Kirchoff stress tensor in material frame
	\begin{itemize}
		\item $\vb{S} := J \vb{F}^{-1} \vb*{\Sigma} \vb{F}^{-T}$
	\end{itemize}
	\item $\vb{P} = \vb{P}(\vb{X}, t)$: first Piola-Kirchoff stress tensor in material frame
	\begin{itemize}
		\item $\vb{P} := J \vb*{\Sigma} \vb{F}^{-T} = \vb{F} \vb{S}$
	\end{itemize}
	\item $\vb{B}, \vb{b}$: body force per unit mass in material/spatial frames
	\item $E, e$: internal energy per unit mass in material frame
	\item $I, i$: potential? energy per unit mass in material frame
	\item $i = i(\vb{X}, t)$: potential? energy per unit mass in spatial frame
	\item $\vb{Q}, \vb{q}$: heat flux in material/spatial frames
\end{itemize}
