\section{Kinematic Variables}

\begin{itemize}
	\item $\vb{U} = \vb{U}(\vb{X}, t)$: displacement in material frame
	\begin{itemize}
		\item $\vb{U} := \vb*{\phi}_{t} - \vb{X}$
	\end{itemize}
	\item $\vb{u} = \vb{u}(\vb{x}, t)$: displacement in spatial frame
	\begin{itemize}
		\item $\vb{u} := \vb{U} \circ \vb*{\phi}_{t}^{-1} = \vb{x} - \vb*{\phi}_{t}^{-1}$
	\end{itemize}
	\item $\vb{V} = \vb{V}(\vb{X}, t)$: velocity in material frame
	\begin{itemize}
		\item $\vb{V} := \pdv{\vb*{\phi}}{t}$
	\end{itemize}
	\item $\vb{v} = \vb{v}(\vb{x}, t)$: velocity in spatial frame
	\begin{itemize}
		\item $\vb{v} := \vb{V} \circ \vb*{\phi}_{t}^{-1} = \pdv{\vb*{\phi}}{t} \circ \vb*{\phi}_{t}^{-1}$
	\end{itemize}
	\item $\vb{L} = \vb{L}(\vb{X}, t)$: velocity gradient in material frame???
	\begin{itemize}
		\item $\vb{L} := \Grad_{\vb{X}} \vb{V}$
	\end{itemize}
	\item $\vb{D} = \vb{D}(\vb{X}, t)$: symmetric component of velocity gradient in material frame???
	\begin{itemize}
		\item $\vb{D} := \frac{1}{2} \left( \vb{L} + \vb{L}^{T} \right)$
	\end{itemize}
	\item $\vb{W} = \vb{W}(\vb{X}, t)$: skew-symmetric component of velocity gradient in material frame???
	\begin{itemize}
		\item $\vb{W} := \frac{1}{2} \left( \vb{L} - \vb{L}^{T} \right)$
	\end{itemize}
\end{itemize}
